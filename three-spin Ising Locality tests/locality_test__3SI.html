# -------------------------------------------------------------------------------------------------------
#######  emulates the action of D1, D2, D3 by using bit-manipulation [X_j = \sigma^x_j operator]  #######
# -------------------------------------------------------------------------------------------------------

def D1_bitmask_pattern(N):                                  # X_A_i . X_B_i   
    pattern = ['1', '1', '0']
    num_cycles = N // len(pattern)
    remaining_elements = N % len(pattern)
    operator_string = pattern * num_cycles + pattern[:remaining_elements]
    return operator_string
def D2_bitmask_pattern(N):                                  # X_B_i . X_C_i
    pattern = ['0', '1', '1']
    num_cycles = N // len(pattern)
    remaining_elements = N % len(pattern)
    operator_string = pattern * num_cycles + pattern[:remaining_elements]
    return operator_string
def D3_bitmask_pattern(N):                                  # X_C_i . X_A_(i+1)
    pattern = ['1', '0', '1']
    num_cycles = N // len(pattern)
    remaining_elements = N % len(pattern)
    operator_string = pattern * num_cycles + pattern[:remaining_elements]
    return operator_string

def symmetry_restricted_basis_of_Ising_nn_3__for_GS(N, sector_blocks = {"D_block__": [0, 0, 0], "k_block__": 0,}):
    @cfunc(op_sig_64,locals=dict(s=int64, n=int64, b=uint64))
    def op(op_struct_ptr, op_str, site_ind, N, args):
        op_struct = carray(op_struct_ptr, 1)[0]
        err = 0
        site_ind = N - site_ind - 1                         # convention for QuSpin for mapping from bits to sites.
        n = (op_struct.state >> site_ind) & 1               # either 0 or 1
        s = (((op_struct.state >> site_ind) & 1) << 1) - 1  # either -1 or 1
        b = 1 << site_ind
        if op_str == 120:                                   # "x" is integer value 120 = ord("x")
            op_struct.state ^= b
        elif op_str == 121:                                 # "y" is integer value 120 = ord("y")
            op_struct.state ^= b
            op_struct.matrix_ele *= 1.0j * s
        elif op_str == 43:                                  # "+" is integer value 43 = ord("+")
            if n:
                op_struct.matrix_ele = 0
            else:
                op_struct.state ^= b                        # create spin
        elif op_str == 45:                                  # "-" is integer value 45 = ord("-")
            if n:
                op_struct.state ^= b                        # destroy spin
            else:
                op_struct.matrix_ele = 0
        elif op_str == 122:                                 # "z" is integer value 120 = ord("z")
            op_struct.matrix_ele *= s
        elif op_str == 110:                                 # "n" is integer value 110 = ord("n")
            op_struct.matrix_ele *= n
        elif op_str == 73:                                  # "I" is integer value 73 = ord("I")
            pass
        else:
            op_struct.matrix_ele = 0
            err = -1
        return err
    op_args = np.array([], dtype=np.uint64)

    # ----------------------------------------------------------------------------------------------
    #######  action of the symmetry maps in terms of integer transformation with bitmasking  #######
    # ----------------------------------------------------------------------------------------------
    @cfunc(map_sig_64,locals=dict(shift=uint64,xmax=uint64,x1=uint64,x2=uint64,period=int64,l=int64))
    def translation(x, N, sign_ptr, args):
        """works for all system sizes N."""
        shift = args[0]                                     # translate state by shift sites
        period = N                                          # periodicity/cyclicity of translation
        xmax = args[1]
        l = (shift + period) % period
        x1 = x >> (period - l)
        x2 = (x << l) & xmax
        return x2 | x1
    T_args = np.array([1, (1 << N) - 1], dtype=np.uint64)
    D1_bitmask = int(''.join(D1_bitmask_pattern(N)), 2)
    D2_bitmask = int(''.join(D2_bitmask_pattern(N)), 2)
    D3_bitmask = int(''.join(D3_bitmask_pattern(N)), 2)
    @cfunc(map_sig_64)
    def D1(x, N, sign_ptr, args):
        """works for all system sizes N."""
        return x ^ D1_bitmask
    D1_args = np.array([], dtype=np.uint64)
    @cfunc(map_sig_64)
    def D2(x, N, sign_ptr, args):
        """works for all system sizes N."""
        return x ^ D2_bitmask
    D2_args = np.array([], dtype=np.uint64)
    @cfunc(map_sig_64)
    def D3(x, N, sign_ptr, args):
        """works for all system sizes N."""
        return x ^ D3_bitmask
    D3_args = np.array([], dtype=np.uint64)

    D1_quantum_num = get_sector_value("D_block__", sector_blocks)[0]
    D2_quantum_num = get_sector_value("D_block__", sector_blocks)[1]
    D3_quantum_num = get_sector_value("D_block__", sector_blocks)[2]

    momentum_quantum_num = get_sector_value("k_block__", sector_blocks)

    maps = dict(
        D1_block=(D1, 2, D1_quantum_num, D1_args),
        D2_block=(D2, 2, D2_quantum_num, D2_args),
        D3_block=(D3, 2, D3_quantum_num, D3_args),
        T_block=(translation, N, momentum_quantum_num, T_args),
        )
    
    op_dict = dict(op=op, op_args=op_args)
    
    return user_basis(
        np.uint64,
        N,
        op_dict,
        allowed_ops=set("+-xyznI"),
        sps=2,
        Ns_block_est=int((2**(N-2))/(N-1)),
        **maps,)




def ground_state__Ising_nn_3__(N, h, fixblock=True, sector_blocks = None):

    if fixblock == True:     
        basis = symmetry_restricted_basis_of_Ising_nn_3__for_GS(N, sector_blocks)
    if fixblock == False:
        basis = spin_basis_1d(N, pauli=True)

    #--------------------------------------------------------------------------------------
    #--------------------------------------------------------------------------------------
    #
    #--- Since we are interested in the entanglement contained in the ground state
    #--- and ground state is known to be in 0 momentum and +1 parity sector, fixing
    #--- ensures that the basis contains only the relevant states, only superposition
    #--- of which produce the subsystems' configurations in the ground state; while
    #--- while eliminating the (nearly) IMPOSSIBLE subsystem-configurations that causes
    #--- vanishing determinant or vanishing eigenvalue of reduced density matrices.
    #--------------------------------------------------------------------------------------
    #--------------------------------------------------------------------------------------
    #--- This is a pathology and it can only be cured for (k-1)-site density matrices for
    #--- h -> 0 since the lack of any entanglement between l-adjacent sites (for all l < k)
    #--- will inevitably cause the divergence of the susceptibility of entanglement 
    #--- entropy as per my calculation.
    #
    #--------------------------------------------------------------------------------------
    #--------------------------------------------------------------------------------------
    
    overall_sign = -1
    
    nn_spin_list = [[overall_sign * 1.0, i, (i+1) % N, (N+2) % N] for i in range(N)]  # Three-spin interaction
    transverse_field_list = [[overall_sign * h, i] for i in range(N)]         # Transverse field
    static = [["zzz", nn_spin_list], ["x", transverse_field_list]]
    dynamic = []                                                # No time-dependent terms
    H = hamiltonian(static, dynamic, basis=basis, dtype=np.float64, check_symm=False, check_herm=False, check_pcon=False)
    _, psi0 = H.eigsh(k=1, which='SA') 
    psi0 = np.sign(psi0[0])*psi0
    return basis, psi0 


def RDM(size, h, RDM_size, fixblock = True, sector_blocks = None):
    if RDM_size <=size:
        basis_ising, ground_state = ground_state__Ising_nn_3__(size, h, fixblock, sector_blocks)
        rdm_ = basis_ising.ent_entropy(state=ground_state, sub_sys_A=[i for i in range(RDM_size)], return_rdm="A")["rdm_A"]
        return rdm_
    else:
        print('RDM_size > system_size : size of reduced density matrix must be smaller than system size')


def Sigma_h__nn_3(size, h, RDM_size, fixblock = True, sector_blocks = None, tolerance=1e-12):
    _rdm = RDM(size, h, RDM_size, fixblock, sector_blocks)
    _d_h_rdm = (1/tolerance)*(RDM(size, h+tolerance, RDM_size, fixblock, sector_blocks) - _rdm)
    squared_d_h_rdm = _d_h_rdm @ _d_h_rdm
    return np.trace(np.linalg.inv(_rdm) @ squared_d_h_rdm).real

def entanglement_entropy__1(size, h, RDM_size, fixblock = True, sector_blocks = None):
    _rdm = RDM(size, h, RDM_size, fixblock, sector_blocks)
    return - np.real(np.trace(_rdm @ scipy.linalg.logm(_rdm)))

def entanglement_entropy__2(size, h, RDM_size, fixblock = True, sector_blocks = None):
    _rdm = RDM(size, h, RDM_size, fixblock, sector_blocks)
    eigEs, _ = scipy.linalg.eigh(_rdm)
    eigEs = eigEs[eigEs > np.finfo(float).eps]
    entropy = 0
    for i in range(len(eigEs)):
        entropy += -eigEs[i] * np.log(eigEs[i])
    return entropy

def entanglement_entropy(size, h, RDM_size, fixblock = True, sector_blocks = None):
    return entanglement_entropy__2(size, h, RDM_size, fixblock = fixblock, sector_blocks = sector_blocks)
